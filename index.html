// 穩定版 Google Sheets 寫入程式（複製全部） function doPost(e) { try { // 強制解析數據（兼容所有格式） let data = {}; if (e.postData.contents.indexOf("data=") > -1) { data = JSON.parse(decodeURIComponent(e.postData.contents.split("data=")[1])); } else { data = JSON.parse(e.postData.contents); } // 驗證必要參數 if (!data.sheetType || !data.record) { return ContentService.createTextOutput(JSON.stringify({ success: false, msg: "缺少必要參數" })).setMimeType(ContentService.MimeType.JSON); } // 獲取試算表（重點：用 ID 確保準確） const ssId = "替換成你的Google Sheets ID"; // ← 這裡要改！ const ss = SpreadsheetApp.openById(ssId); const sheetName = data.sheetType === "customer" ? "客人" : "產品"; const sheet = ss.getSheetByName(sheetName); if (!sheet) { return ContentService.createTextOutput(JSON.stringify({ success: false, msg: "找不到工作表：" + sheetName + "，請確認工作表名稱正確" })).setMimeType(ContentService.MimeType.JSON); } // 處理日期 const dateParts = data.record.date ? data.record.date.split("-") : ["", "", ""]; // 準備寫入數據（嚴格對應 A-I 欄） const row = [ dateParts[0] || "", // A:年 dateParts[1] || "", // B:月 dateParts[2] || "", // C:日 data.record.staff ? data.record.staff.charAt(0).toUpperCase() : "", // D:首字母 data.record.staff || "", // E:員工名字 data.record.staffId || "",// F:工號 data.record.items || "", // G:產品 data.record.price || "", // H:價錢 data.record.payment || "" // I:支付方式 ]; // 寫入數據（強制在第一行後插入） sheet.appendRow(row); // 返回成功 return ContentService.createTextOutput(JSON.stringify({ success: true, msg: "數據已成功寫入 Google Sheets", row: row })).setMimeType(ContentService.MimeType.JSON); } catch (error) { // 詳細錯誤信息 return ContentService.createTextOutput(JSON.stringify({ success: false, msg: "錯誤詳情：" + error.toString(), line: error.lineNumber || "未知行數" })).setMimeType(ContentService.MimeType.JSON); } } // 測試接口（用於驗證） function doGet() { return ContentService.createTextOutput("API 運行正常！當前時間：" + new Date().toString()) .setMimeType(ContentService.MimeType.TEXT); } // 手動測試寫入（用於調試） function testWriteData() { const ss = SpreadsheetApp.openById("替換成你的Google Sheets ID"); const sheet = ss.getSheetByName("產品"); sheet.appendRow(["2026", "03", "22", "A", "員工A", "A001", "紅罐x1", "178", "現金"]); console.log("測試數據已寫入"); }
